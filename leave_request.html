▶ 📋 Supabase SQL (테이블 생성)
-- 연장근무 보고서
CREATE TABLE IF NOT EXISTS overtime_reports (
  id bigint generated always as identity primary key,
  today_date date, dept text, name text, position text,
  prev_date date, prev_normal text, prev_overtime text, prev_note text,
  today_normal text, today_overtime text, today_note text,
  holiday_date date, holiday_work text, holiday_note text,
  approval text default '', created_at timestamptz default now()
);
-- 사원
CREATE TABLE IF NOT EXISTS employees (
  id bigint generated always as identity primary key,
  name text, dept text, position text, sort_order int default 0,
  hire_date date, created_at timestamptz default now()
);
-- RLS
ALTER TABLE overtime_reports ENABLE ROW LEVEL SECURITY;
ALTER TABLE employees ENABLE ROW LEVEL SECURITY;
DROP POLICY IF EXISTS "anon_all" ON overtime_reports;
CREATE POLICY "anon_all" ON overtime_reports FOR ALL TO anon USING (true) WITH CHECK (true);
DROP POLICY IF EXISTS "anon_all" ON employees;
CREATE POLICY "anon_all" ON employees FOR ALL TO anon USING (true) WITH CHECK (true);
