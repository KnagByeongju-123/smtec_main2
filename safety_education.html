CREATE TABLE IF NOT EXISTS safety_education_log (
  id BIGSERIAL PRIMARY KEY,
  edu_date DATE NOT NULL,
  edu_time TIME,
  work_name TEXT NOT NULL,
  vendor TEXT,
  head_count INT,
  work_location TEXT,
  risks TEXT,
  edu_items TEXT,
  edu_count INT,
  edu_total INT,
  instructor TEXT,
  attendees TEXT,
  writer TEXT,
  signature TEXT,
  memo TEXT,
  result TEXT,
  created_at TIMESTAMPTZ DEFAULT NOW()
);
