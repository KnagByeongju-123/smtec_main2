▶ 📋 Supabase SQL (테이블 생성)
-- 기종 마스터
CREATE TABLE IF NOT EXISTS defect_model (
  id bigint generated always as identity primary key,
  model_name text not null, customer text,
  category1 text, category2 text,
  created_at timestamptz default now()
);
-- 기판 이미지
CREATE TABLE IF NOT EXISTS board_image (
  id bigint generated always as identity primary key,
  model_id bigint references defect_model(id),
  side text default 'TOP', image_url text,
  created_at timestamptz default now()
);
-- 영역
CREATE TABLE IF NOT EXISTS defect_zone (
  id bigint generated always as identity primary key,
  model_id bigint references defect_model(id),
  side text default 'TOP', name text,
  x_pct numeric, y_pct numeric,
  w_pct numeric, h_pct numeric,
  created_at timestamptz default now()
);
-- 불량유형
CREATE TABLE IF NOT EXISTS defect_type (
  id bigint generated always as identity primary key,
  name text not null, category text,
  sort_order integer default 0,
  created_at timestamptz default now()
);
-- 불량이력
CREATE TABLE IF NOT EXISTS defect_history (
  id bigint generated always as identity primary key,
  model_id bigint, model_name text,
  side text, zone_name text,
  defect_type text, defect_category text,
  qty integer default 0, inspect_qty integer,
  worker text, inspect_line text,
  prod_line text, note text,
  recorded_at timestamptz default now()
);
-- RLS
ALTER TABLE defect_model ENABLE ROW LEVEL SECURITY;
ALTER TABLE board_image ENABLE ROW LEVEL SECURITY;
ALTER TABLE defect_zone ENABLE ROW LEVEL SECURITY;
ALTER TABLE defect_type ENABLE ROW LEVEL SECURITY;
ALTER TABLE defect_history ENABLE ROW LEVEL SECURITY;
DO $$ BEGIN
  EXECUTE 'DROP POLICY IF EXISTS "anon_all" ON defect_model; CREATE POLICY "anon_all" ON defect_model FOR ALL TO anon USING (true) WITH CHECK (true)';
  EXECUTE 'DROP POLICY IF EXISTS "anon_all" ON board_image; CREATE POLICY "anon_all" ON board_image FOR ALL TO anon USING (true) WITH CHECK (true)';
  EXECUTE 'DROP POLICY IF EXISTS "anon_all" ON defect_zone; CREATE POLICY "anon_all" ON defect_zone FOR ALL TO anon USING (true) WITH CHECK (true)';
  EXECUTE 'DROP POLICY IF EXISTS "anon_all" ON defect_type; CREATE POLICY "anon_all" ON defect_type FOR ALL TO anon USING (true) WITH CHECK (true)';
  EXECUTE 'DROP POLICY IF EXISTS "anon_all" ON defect_history; CREATE POLICY "anon_all" ON defect_history FOR ALL TO anon USING (true) WITH CHECK (true)';
END $$;
