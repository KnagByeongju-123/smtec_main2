▶ 📋 Supabase SQL (테이블 생성)
CREATE TABLE IF NOT EXISTS qc_machine_check (
  id bigint generated always as identity primary key,
  check_date date not null, check_time time not null,
  inspector text not null, machine text not null, qr_code text,
  chk_input text default 'OK', chk_chain text default 'OK',
  chk_height text default 'OK', chk_camera text default 'OK',
  chk_belt text default 'OK', chk_plate text default 'OK',
  chk_eddy text default 'OK', chk_out text default 'OK',
  chk_sample text default 'OK',
  ng_count integer default 0, result text default 'OK',
  memo text, created_at timestamptz default now()
);
ALTER TABLE qc_machine_check ENABLE ROW LEVEL SECURITY;
DROP POLICY IF EXISTS "anon_all" ON qc_machine_check;
CREATE POLICY "anon_all" ON qc_machine_check FOR ALL TO anon USING (true) WITH CHECK (true);
