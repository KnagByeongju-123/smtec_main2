📋 Supabase SQL (테이블 생성) 
CREATE TABLE IF NOT EXISTS shipping_check (
  id BIGSERIAL PRIMARY KEY,
  check_date DATE NOT NULL,
  check_time TIME NOT NULL,
  line TEXT NOT NULL,
  inspector TEXT NOT NULL,
  results JSONB NOT NULL,
  ng_count INT DEFAULT 0,
  judgement TEXT,
  memo TEXT,
  photos JSONB,
  created_at TIMESTAMPTZ DEFAULT NOW()
);
ALTER TABLE shipping_check DISABLE ROW LEVEL SECURITY;
GRANT ALL ON shipping_check TO anon, authenticated;
GRANT USAGE, SELECT ON SEQUENCE shipping_check_id_seq TO anon, authenticated;
