▶ 📋 Supabase SQL (테이블 생성)
CREATE TABLE IF NOT EXISTS shipment_history (
  id bigint generated always as identity primary key,
  reg_date date, part_number text, qty integer,
  lot_number text, box_count integer, per_box integer,
  customer text, note text,
  created_at timestamptz default now()
);
CREATE TABLE IF NOT EXISTS box_unit (
  id bigint generated always as identity primary key,
  part_number text not null, per_box integer default 1,
  customer text, is_active boolean default true,
  created_at timestamptz default now()
);
CREATE TABLE IF NOT EXISTS work_items (
  id bigint generated always as identity primary key,
  part_number text, qty integer, lot_number text,
  box_count integer, per_box integer, customer text,
  created_at timestamptz default now()
);
-- RLS
ALTER TABLE shipment_history ENABLE ROW LEVEL SECURITY;
ALTER TABLE box_unit ENABLE ROW LEVEL SECURITY;
ALTER TABLE work_items ENABLE ROW LEVEL SECURITY;
DROP POLICY IF EXISTS "anon_all" ON shipment_history;
CREATE POLICY "anon_all" ON shipment_history FOR ALL TO anon USING(true) WITH CHECK(true);
DROP POLICY IF EXISTS "anon_all" ON box_unit;
CREATE POLICY "anon_all" ON box_unit FOR ALL TO anon USING(true) WITH CHECK(true);
DROP POLICY IF EXISTS "anon_all" ON work_items;
CREATE POLICY "anon_all" ON work_items FOR ALL TO anon USING(true) WITH CHECK(true);
