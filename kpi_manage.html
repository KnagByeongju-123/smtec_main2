* Supabase 테이블 생성 후 사용 (SQL 아래 참조)

📄 Supabase SQL (테이블 생성) 
-- 프로세스 마스터
CREATE TABLE IF NOT EXISTS kpi_process (
  id SERIAL PRIMARY KEY,
  name TEXT NOT NULL UNIQUE,
  sort_order INT DEFAULT 0,
  created_at TIMESTAMPTZ DEFAULT NOW()
);

-- KPI 마스터
CREATE TABLE IF NOT EXISTS kpi_master (
  id SERIAL PRIMARY KEY,
  process_id INT REFERENCES kpi_process(id),
  kpi_name TEXT NOT NULL,
  unit TEXT DEFAULT '%',
  direction TEXT DEFAULT 'higher_better',
  target_value NUMERIC,
  alert_value NUMERIC,
  iatf_clause TEXT,
  dept TEXT,
  description TEXT,
  frequency TEXT DEFAULT 'monthly',
  sort_order INT DEFAULT 0,
  is_active BOOLEAN DEFAULT TRUE,
  created_at TIMESTAMPTZ DEFAULT NOW()
);

-- 기존 테이블에 dept 컬럼 추가 (이미 테이블이 있는 경우)
-- ALTER TABLE kpi_master ADD COLUMN IF NOT EXISTS dept TEXT;

-- 월별 실적
CREATE TABLE IF NOT EXISTS kpi_record (
  id SERIAL PRIMARY KEY,
  kpi_id INT REFERENCES kpi_master(id),
  year INT NOT NULL,
  month INT NOT NULL,
  actual_value NUMERIC,
  target_value NUMERIC,
  note TEXT,
  created_at TIMESTAMPTZ DEFAULT NOW(),
  UNIQUE(kpi_id, year, month)
);

-- RLS
ALTER TABLE kpi_process ENABLE ROW LEVEL SECURITY;
ALTER TABLE kpi_master ENABLE ROW LEVEL SECURITY;
ALTER TABLE kpi_record ENABLE ROW LEVEL SECURITY;
DROP POLICY IF EXISTS "kpi_process_all" ON kpi_process;
CREATE POLICY "kpi_process_all" ON kpi_process FOR ALL USING (true) WITH CHECK (true);
DROP POLICY IF EXISTS "kpi_master_all" ON kpi_master;
CREATE POLICY "kpi_master_all" ON kpi_master FOR ALL USING (true) WITH CHECK (true);
DROP POLICY IF EXISTS "kpi_record_all" ON kpi_record;
CREATE POLICY "kpi_record_all" ON kpi_record FOR ALL USING (true) WITH CHECK (true);
