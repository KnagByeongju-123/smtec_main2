▶ 📋 Supabase SQL (테이블 생성)
CREATE TABLE IF NOT EXISTS p_machine_check (
  id bigint generated always as identity primary key,
  check_date date not null, check_time text,
  inspector text not null, machine text not null,
  qr_overload text, qr_motor text,
  chk_overload text default 'OK', chk_balance text default 'OK',
  chk_clutchbrake text default 'OK', chk_motor text default 'OK',
  chk_clutch text default 'OK', chk_safety text default 'OK',
  chk_oil text default 'OK', chk_pitch text default 'OK',
  chk_helmet text default 'OK',
  ng_count integer default 0, result text default 'OK',
  memo text, created_at timestamptz default now()
);
ALTER TABLE p_machine_check ENABLE ROW LEVEL SECURITY;
DROP POLICY IF EXISTS "anon_all" ON p_machine_check;
CREATE POLICY "anon_all" ON p_machine_check
  FOR ALL TO anon USING (true) WITH CHECK (true);

CREATE TABLE IF NOT EXISTS p_machine_group_state (
  maker text primary key,
  is_open boolean default false,
  updated_at timestamptz default now()
);
ALTER TABLE p_machine_group_state ENABLE ROW LEVEL SECURITY;
DROP POLICY IF EXISTS "anon_all" ON p_machine_group_state;
CREATE POLICY "anon_all" ON p_machine_group_state
  FOR ALL TO anon USING (true) WITH CHECK (true);
