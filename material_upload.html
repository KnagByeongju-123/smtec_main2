▶ 📋 Supabase SQL (테이블 생성)
CREATE TABLE IF NOT EXISTS move_history_tj (
  id bigint generated always as identity primary key,
  move_date date, move_time text, process text,
  departure text, destination text,
  barcode_raw text, base_part text, lot text,
  qty integer, note text, format text,
  worker text, equipment text, weight numeric,
  incoming_date text, parent_part text,
  child_type text, model_type text,
  general_model text, material_info text,
  created_at timestamptz default now()
);
ALTER TABLE move_history_tj ENABLE ROW LEVEL SECURITY;
DROP POLICY IF EXISTS "anon_all" ON move_history_tj;
CREATE POLICY "anon_all" ON move_history_tj FOR ALL TO anon USING (true) WITH CHECK (true);
